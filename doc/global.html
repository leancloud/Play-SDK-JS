    constant Event :string

    事件
    Properties:
    Name Type Description
    CONNECTED string 连接成功
    CONNECT_FAILED string 连接失败
    DISCONNECTED string 断开连接
    JOINED_LOBBY string 加入到大厅
    LEFT_LOBBY string 离开大厅
    LOBBY_ROOM_LIST_UPDATE string 大厅房间列表变化
    CREATED_ROOM string 创建房间成功
    CREATE_ROOM_FAILED string 创建房间失败
    JOINED_ROOM string 加入房间成功
    JOIN_ROOM_FAILED string 加入房间失败
    NEW_PLAYER_JOINED_ROOM string 有新玩家加入房间
    PLAYER_LEFT_ROOM string 有玩家离开房间
    PLAYER_ACTIVITY_CHANGED string 玩家活跃属性变化
    MASTER_SWITCHED string 主机变更
    LEFT_ROOM string 离开房间
    ROOM_CUSTOM_PROPERTIES_CHANGED string 房间自定义属性变化
    PLAYER_CUSTOM_PROPERTIES_CHANGED string 玩家自定义属性变化
    CUSTOM_EVENT string 自定义事件
    ERROR string 错误事件
